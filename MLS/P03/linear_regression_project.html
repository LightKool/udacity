""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if len(A) != len(b):
        return None
    M = augmentMatrix(A,b)
    rows, cols = shape(M)
    matxRound(M, decPts)
    for row in range(rows):
        maxrow = row
        for row2 in range(row+1, rows):
            if abs(M[row2][row]) > abs(M[maxrow][row]):
                maxrow = row2
        swapRows(M, row, maxrow)
        if abs(M[row][row]) < epsilon:
            return None
        scaleRow(M, row, 1/M[row][row])
        for row2 in (range(row) + range(row+1,rows)):
            scale = -M[row2][row]/M[row][row]
            addScaledRow(M, row2, row, scale)
    result = [[x] for x in [r[cols-1] for r in M]]
    return result
